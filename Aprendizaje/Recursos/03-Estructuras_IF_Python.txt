class SistemaTienda:
    def __init__(self):
        self.inventario = {"laptop": 5, "mouse": 10, "teclado": 3}
        self.descuentos = {"VIP": 0.20, "PREMIUM": 0.15, "BASIC": 0.05}
    
    def procesar_compra(self, producto, cantidad, tipo_cliente, saldo_cliente):
        print(f"\nğŸ›’ PROCESANDO COMPRA: {cantidad}x {producto}")
        print(f"ğŸ‘¤ Tipo de cliente: {tipo_cliente}")
        
        # Verificar producto en inventario
        if producto not in self.inventario:
            print(f"âŒ PRODUCTO NO DISPONIBLE: {producto}")
            return False
            
        # Verificar stock suficiente
        if self.inventario[producto] < cantidad:
            print(f"âŒ STOCK INSUFICIENTE: Solo quedan {self.inventario[producto]} unidades")
            return False
        
        # Calcular precio y descuento
        precios = {"laptop": 1000, "mouse": 25, "teclado": 75}
        precio_unitario = precios[producto]
        descuento = self.descuentos.get(tipo_cliente, 0)
        
        total = precio_unitario * cantidad * (1 - descuento)
        
        # Verificar fondos
        if saldo_cliente >= total:
            self.inventario[producto] -= cantidad
            saldo_cliente -= total
            print(f"âœ… COMPRA EXITOSA!")
            print(f"ğŸ’° Total: ${total:.2f} (Descuento: {descuento*100}%)")
            print(f"ğŸ’³ Saldo restante: ${saldo_cliente:.2f}")
            print(f"ğŸ“¦ Stock actual de {producto}: {self.inventario[producto]}")
            return True
        else:
            print(f"âŒ FONDOS INSUFICIENTES")
            print(f"ğŸ’¸ Necesitas: ${total:.2f}, Tienes: ${saldo_cliente:.2f}")
            return False
