class SistemaTienda:
    def __init__(self):
        self.inventario = {"laptop": 5, "mouse": 10, "teclado": 3}
        self.descuentos = {"VIP": 0.20, "PREMIUM": 0.15, "BASIC": 0.05}
    
    def procesar_compra(self, producto, cantidad, tipo_cliente, saldo_cliente):
        print(f"\n🛒 PROCESANDO COMPRA: {cantidad}x {producto}")
        print(f"👤 Tipo de cliente: {tipo_cliente}")
        
        # Verificar producto en inventario
        if producto not in self.inventario:
            print(f"❌ PRODUCTO NO DISPONIBLE: {producto}")
            return False
            
        # Verificar stock suficiente
        if self.inventario[producto] < cantidad:
            print(f"❌ STOCK INSUFICIENTE: Solo quedan {self.inventario[producto]} unidades")
            return False
        
        # Calcular precio y descuento
        precios = {"laptop": 1000, "mouse": 25, "teclado": 75}
        precio_unitario = precios[producto]
        descuento = self.descuentos.get(tipo_cliente, 0)
        
        total = precio_unitario * cantidad * (1 - descuento)
        
        # Verificar fondos
        if saldo_cliente >= total:
            self.inventario[producto] -= cantidad
            saldo_cliente -= total
            print(f"✅ COMPRA EXITOSA!")
            print(f"💰 Total: ${total:.2f} (Descuento: {descuento*100}%)")
            print(f"💳 Saldo restante: ${saldo_cliente:.2f}")
            print(f"📦 Stock actual de {producto}: {self.inventario[producto]}")
            return True
        else:
            print(f"❌ FONDOS INSUFICIENTES")
            print(f"💸 Necesitas: ${total:.2f}, Tienes: ${saldo_cliente:.2f}")
            return False

# ¡USO DEL SISTEMA!
tienda = SistemaTienda()
tienda.procesar_compra("laptop", 2, "VIP", 2000)
tienda.procesar_compra("mouse", 15, "BASIC", 100)  # Stock insuficiente

def sistema_de_juego(vida, energia, armas, enemigos_cerca):
    print("\n🎮 ¡SISTEMA DE ESTADO DEL JUGADOR! 🎮")
    
    # Evaluar estado del jugador
    if vida <= 0:
        print("💀 ¡GAME OVER! - Vida agotada")
        return "game_over"
    
    elif vida < 20:
        print("🔴 ¡POCA VIDA! - Busca un kit de salud")
        if enemigos_cerca:
            print("⚠️ ¡ENEMIGOS CERCA! - Considera huir")
    
    elif vida < 50:
        print("🟡 VIDA MEDIA - Ten cuidado")
    
    else:
        print("🟢 VIDA ALTA - Estás en buena forma")
    
    # Evaluar energía
    if energia < 10:
        print("😴 ¡SIN ENERGÍA! - Descansa o busca comida")
    elif energia < 30:
        print("🟡 Energía baja - Limita acciones especiales")
    
    # Evaluar armas
    if not armas:
        print("👊 Sin armas - Solo combate cuerpo a cuerpo")
        if enemigos_cerca:
            print("🎯 ¡RECOMENDACIÓN! - Busca un arma rápido")
    elif "escopeta" in armas or "rifle" in armas:
        print("🔫 Bien armado - Puedes enfrentar múltiples enemigos")
    else:
        print("🔪 Arma básica - Eficaz a corta distancia")
    
    # Decisión final
    if vida > 70 and len(armas) >= 2 and energia > 50:
        print("🚀 ¡ESTADO ÓPTIMO! - Avanza con confianza")
        return "avanzar"
    elif vida < 30 and enemigos_cerca:
        print("🏃 ¡RETIRADA RECOMENDADA! - Busca refuerzos")
        return "retirar"
    else:
        print("🛡️ Estado estable - Procede con precaución")
        return "explorar"

# ¡SIMULACIÓN DE JUEGO!
estado = sistema_de_juego(
    vida=45, 
    energia=25, 
    armas=["pistola", "cuchillo"], 
    enemigos_cerca=True
)

    🎯 if - Ejecuta código si la condición es True

    ⚡ Ternario - if-else en una sola línea

¡OPERADORES CLAVE!
