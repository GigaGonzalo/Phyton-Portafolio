class CalculadoraAvanzada:
    """Calculadora que demuestra todos los operadores de Python"""
    
    def __init__(self):
        self.historial = []
    
    def operaciones_aritmeticas(self, a, b):
        print("ğŸ§® OPERACIONES ARITMÃ‰TICAS:")
        resultados = {
            "Suma": a + b,
            "Resta": a - b,
            "MultiplicaciÃ³n": a * b,
            "DivisiÃ³n": a / b if b != 0 else "Indefinido",
            "DivisiÃ³n Entera": a // b if b != 0 else "Indefinido",
            "MÃ³dulo": a % b if b != 0 else "Indefinido",
            "Potencia": a ** b
        }
        
        for operacion, resultado in resultados.items():
            print(f"   {operacion}: {a} op {b} = {resultado}")
        
        return resultados
    
    def operaciones_comparacion(self, a, b):
        print("\nğŸ” OPERACIONES DE COMPARACIÃ“N:")
        comparaciones = {
            "Igualdad": a == b,
            "Desigualdad": a != b,
            "Mayor que": a > b,
            "Menor que": a < b,
            "Mayor o igual": a >= b,
            "Menor o igual": a <= b
        }
        
        for comparacion, resultado in comparaciones.items():
            print(f"   {a} {comparacion.lower()} {b}? {resultado}")
        
        return comparaciones
    
    def operaciones_logicas(self, cond1, cond2):
        print("\nğŸ§  OPERACIONES LÃ“GICAS:")
        logicas = {
            "AND": cond1 and cond2,
            "OR": cond1 or cond2,
            "NOT cond1": not cond1,
            "NOT cond2": not cond2
        }
        
        for operacion, resultado in logicas.items():
            print(f"   {operacion}: {resultado}")
        
        return logicas
    
    def operaciones_bits(self, a, b):
        print("\nğŸ”¢ OPERACIONES A NIVEL DE BITS:")
        bits = {
            "AND": a & b,
            "OR": a | b,
            "XOR": a ^ b,
            "NOT a": ~a,
            "Desplazamiento izquierda a << 1": a << 1,
            "Desplazamiento derecha a >> 1": a >> 1
        }
        
        for operacion, resultado in bits.items():
            print(f"   {operacion}: {resultado} ({bin(resultado)})")
        
        return bits
    
    def demostrar_precedencia(self):
        print("\nğŸ“Š DEMOSTRACIÃ“N DE PRECEDENCIA:")
        ejemplos = [
            ("5 + 3 * 2", 5 + 3 * 2),
            ("(5 + 3) * 2", (5 + 3) * 2),

        ]
        
        for expresion, resultado in ejemplos:
