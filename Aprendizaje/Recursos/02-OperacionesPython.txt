🛒 SISTEMA DE CARRITO DE COMPRAS


class CarritoCompras:
    def __init__(self):
        self.productos = []
        self.total = 0.0
    
    def agregar_producto(self, nombre, precio, cantidad=1):
        self.productos.append({"nombre": nombre, "precio": precio, "cantidad": cantidad})
        self.total += precio * cantidad
        print(f"✅ {cantidad}x {nombre} agregado - Subtotal: ${precio * cantidad:.2f}")
    
    def calcular_descuento(self, porcentaje_descuento):
        descuento = self.total * (porcentaje_descuento / 100)
        total_con_descuento = self.total - descuento
        print(f"🎁 Descuento del {porcentaje_descuento}%: -${descuento:.2f}")
        print(f"💰 TOTAL FINAL: ${total_con_descuento:.2f}")
        return total_con_descuento

# ¡USO EN TIENDA REAL!
carrito = CarritoCompras()
carrito.agregar_producto("Laptop", 999.99, 1)
carrito.agregar_producto("Mouse", 25.50, 2)
carrito.agregar_producto("Teclado", 75.00, 1)
carrito.calcular_descuento(10)
