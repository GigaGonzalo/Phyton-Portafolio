def procesador_datos(datos):
    """FunciÃ³n principal que contiene funciones auxiliares"""
    
    # Â¡FUNCIONES ANIDADAS - SOLO VISIBLES DENTRO!
    def validar_dato(dato):
        """Valida que el dato sea numÃ©rico"""
        return isinstance(dato, (int, float))
    
    def limpiar_dato(dato):
        """Limpia y formatea el dato"""
        if isinstance(dato, str):
            try:
                return float(dato)
            except ValueError:
                return None
        return dato
    
    def calcular_estadisticas(numeros):
        """Calcula estadÃ­sticas bÃ¡sicas"""
        if not numeros:
            return None
        
        return {
            "promedio": sum(numeros) / len(numeros),
            "maximo": max(numeros),
            "minimo": min(numeros),
            "total": sum(numeros)
        }
    
    # Â¡PROCESAMIENTO PRINCIPAL!
    print("ğŸ” PROCESANDO DATOS...")
    
    datos_limpios = []
    for dato in datos:
        dato_limpio = limpiar_dato(dato)
        if validar_dato(dato_limpio):
            datos_limpios.append(dato_limpio)
        else:
            print(f"   ğŸ—‘ï¸ Dato invÃ¡lido descartado: {dato}")
    
    estadisticas = calcular_estadisticas(datos_limpios)
    
    if estadisticas:
        print("ğŸ“Š ESTADÃSTICAS CALCULADAS:")
        for clave, valor in estadisticas.items():
            print(f"   {clave.capitalize()}: {valor:.2f}")
    else:
        print("âŒ No hay datos vÃ¡lidos para analizar")
    
    return estadisticas

# Â¡DECORADORES - FUNCIONES QUE MEJORAN OTRAS FUNCIONES!
def decorador_tiempo(funcion):
    """Decorador que mide el tiempo de ejecuciÃ³n"""
    def wrapper(*args, **kwargs):
        import time
        inicio = time.time()
        resultado = funcion(*args, **kwargs)
        fin = time.time()
        print(f"â±ï¸ {funcion.__name__} tardÃ³ {fin - inicio:.4f} segundos")
        return resultado
    return wrapper

def decorador_log(funcion):
    """Decorador que registra llamadas a funciones"""
    def wrapper(*args, **kwargs):
        print(f"ğŸ“ Llamando a {funcion.__name__} con args: {args}, kwargs: {kwargs}")
        resultado = funcion(*args, **kwargs)
        print(f"âœ… {funcion.__name__} retornÃ³: {resultado}")
        return resultado
    return wrapper

# Â¡APLICANDO DECORADORES!
@decorador_tiempo
@decorador_log
def calcular_factorial(n):
    """Calcula el factorial de un nÃºmero"""
    if n == 0 or n == 1:
        return 1
    resultado = 1
    for i in range(2, n + 1):
        resultado *= i
    return resultado

@decorador_tiempo
def fibonacci(n):
    """Calcula el n-Ã©simo nÃºmero de Fibonacci"""
    if n <= 1:
        return n
    a, b = 0, 1
    for _ in range(2, n + 1):
        a, b = b, a + b
    return b

print("ğŸ© Â¡FUNCIONES DECORADAS EN ACCIÃ“N! ğŸ©")
print(f"ğŸ”¢ Factorial de 5: {calcular_factorial(5)}")
print(f"ğŸ”¢ Fibonacci de 10: {fibonacci(10)}")

# Â¡DECORADOR CON ARGUMENTOS!
def repetir(n_veces):
    """Decorador que repite la ejecuciÃ³n de una funciÃ³n"""
    def decorador_real(funcion):
        def wrapper(*args, **kwargs):
            resultados = []
            for i in range(n_veces):
                print(f"ğŸ”„ EjecuciÃ³n {i + 1}/{n_veces}")
                resultado = funcion(*args, **kwargs)
                resultados.append(resultado)
            return resultados
        return wrapper
    return decorador_real

class SistemaEcommerce:
    def __init__(self):
        self.carrito = []
        self.inventario = {
            "laptop": {"precio": 1000, "stock": 5},
            "mouse": {"precio": 25, "stock": 10},
            "teclado": {"precio": 75, "stock": 3}
        }
    
    # Â¡FUNCIONES DE VALIDACIÃ“N!
    def validar_producto(self, producto):
        return producto in self.inventario
    
    def validar_stock(self, producto, cantidad):
        return self.inventario[producto]["stock"] >= cantidad
    
    # Â¡FUNCIONES DE CÃLCULO!
    def calcular_subtotal(self):
        return sum(item["precio"] * item["cantidad"] for item in self.carrito)
    
    def calcular_iva(self, subtotal, porcentaje=16):
        return subtotal * (porcentaje / 100)
    
    def calcular_total(self, subtotal, iva):
        return subtotal + iva
    
    # Â¡FUNCIONES DE OPERACIÃ“N!
    def agregar_al_carrito(self, producto, cantidad=1):
        if not self.validar_producto(producto):
            print(f"âŒ Producto '{producto}' no existe")
            return False
        
        if not self.validar_stock(producto, cantidad):
            print(f"âŒ Stock insuficiente de '{producto}'")
            return False
        
        precio = self.inventario[producto]["precio"]
        self.carrito.append({
            "producto": producto,
            "precio": precio,
            "cantidad": cantidad
        })
        
        self.inventario[producto]["stock"] -= cantidad
        print(f"âœ… {cantidad}x '{producto}' agregado al carrito")
        return True
    
    def procesar_compra(self):
        if not self.carrito:
            print("âŒ Carrito vacÃ­o")
            return False
        
        # Â¡USO DE MÃšLTIPLES FUNCIONES!
        subtotal = self.calcular_subtotal()
        iva = self.calcular_iva(subtotal)
        total = self.calcular_total(subtotal, iva)
        
        # Mostrar resumen
        self.mostrar_resumen_compra(subtotal, iva, total)
        
        # Confirmar compra
        if self.confirmar_compra(total):
            print("ğŸ‰ Â¡COMPRA EXITOSA!")
            self.carrito = []  # Vaciar carrito
            return True
        else:
            print("âŒ Compra cancelada")
            return False
    
    # Â¡FUNCIONES DE PRESENTACIÃ“N!
    def mostrar_resumen_compra(self, subtotal, iva, total):
        print("\nğŸ“‹ RESUMEN DE COMPRA:")
        for item in self.carrito:
            print(f"   {item['cantidad']}x {item['producto']} - ${item['precio']} c/u")
        
        print(f"\nğŸ’° Subtotal: ${subtotal:.2f}")
        print(f"ğŸ“Š IVA (16%): ${iva:.2f}")
        print(f"ğŸ’³ Total: ${total:.2f}")
    
    def confirmar_compra(self, total):
        respuesta = input(f"\nÂ¿Confirmar compra de ${total:.2f}? (s/n): ")
        return respuesta.lower() == 's'
    
    def mostrar_inventario(self):
        print("\nğŸ“¦ INVENTARIO DISPONIBLE:")
        for producto, info in self.inventario.items():
            print(f"   {producto}: ${info['precio']} (Stock: {info['stock']})")
