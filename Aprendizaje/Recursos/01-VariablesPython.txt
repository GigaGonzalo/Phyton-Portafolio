🎁 ¡BONUS: VARIABLES GLOBALES VS LOCALES!


# ¡VARIABLE GLOBAL - ACCESIBLE EN TODOS LADOS!
contador_global = 0

def incrementar_contador():
    # ¡VARIABLE LOCAL - SOLO EN ESTA FUNCIÓN!
    contador_local = 0
    global contador_global
    
    contador_local += 1
    contador_global += 1
    
    print(f"Local: {contador_local}, Global: {contador_global}")

# ¡PRUEBA LA DIFERENCIA!
incrementar_contador()
incrementar_contador()
